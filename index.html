function checkSvg(){


  return '<svg viewBox="0 0 24 24" fill="none" stroke="#0a0908" stroke-width="3.2" stroke-linecap="round" stroke-linejoin="round"><polyline points="20 6 9 17 4 12"></polyline></svg>';


}


 


function computeCounts(items){


  const done = items.filter(i => i.d).length;


  return { done, total: items.length, pct: items.length ? Math.round(done/items.length*100) : 0 };


}


 


function deptCounts(dept){


  let done = 0, total = 0;


  dept.groups.forEach(g => { done += g.items.filter(i=>i.d).length; total += g.items.length; });


  return { done, total, pct: total ? Math.round(done/total*100) : 0 };


}


 


function renderTabs(){


  const tabs = document.getElementById('tabs');


  tabs.innerHTML = '';


  data.departments.forEach((dept, di) => {


    const { pct } = deptCounts(dept);


    const el = document.createElement('div');


    el.className = 'tab' + (di === activeDept ? ' active' : '');


    el.innerHTML = `<span>${dept.name}</span><span class="pct">${pct}%</span><span class="tab-x" data-di="${di}">&#10005;</span>`;


    el.addEventListener('click', (e) => {


      if(e.target.classList.contains('tab-x')) return;


      activeDept = di;


      render();


    });


    el.querySelector('.tab-x').addEventListener('click', (e) => {


      e.stopPropagation();


      if(data.departments.length <= 1) return;


      if(!confirm(`Remove the "${dept.name}" department and all its tasks?`)) return;


      data.departments.splice(di, 1);


      if(activeDept >= data.departments.length) activeDept = data.departments.length - 1;


      render(); save();


    });

  });

    input.value = data.title;


    display.style.display = 'none';


    input.style.display = 'inline-block';


    input.focus();


    input.select();


  }


  function commit(){


    const val = input.value.trim();


    if(val) data.title = val;


    display.textContent = data.title;


    display.style.display = 'inline-block';


    input.style.display = 'none';


    save();


  }


  editBtn.addEventListener('click', enterEdit);


  display.addEventListener('click', enterEdit);


  input.addEventListener('keydown', (e) => { if(e.key === 'Enter') commit(); });


  input.addEventListener('blur', commit);


}

function addDepartment(){


  const input = document.getElementById('newDeptInput');


  const val = input.value.trim();


  if(!val) return;


  data.departments.push({ name: val, groups: [{ name: 'Unassigned', items: [] }] });


  activeDept = data.departments.length - 1;


  input.value = '';


  document.getElementById('newDeptRow').style.display = 'none';


  render(); save();


}


 


async function save(){


  try{


    await withTimeout(window.storage.set(STORAGE_KEY, JSON.stringify(data), true), 5000);


  }catch(e){


    console.error('Save failed (working offline):', e);


  }


}


 


function withTimeout(promise, ms){


  return Promise.race([


    promise,


    new Promise((_, reject) => setTimeout(() => reject(new Error('storage timeout')), ms))


  ]);


}


 


async function load(){


  let usingStorage = true;


  try{


    const result = await withTimeout(window.storage.get(STORAGE_KEY, true), 5000);


    if(result && result.value){


      data = JSON.parse(result.value);


    }else{


      // attempt migration from old media-only tracker


      let migratedGroups = null;


      try{


        const old = await withTimeout(window.storage.get(OLD_KEY, true), 5000);


        if(old && old.value) migratedGroups = JSON.parse(old.value);


      }catch(e){ /* no old data, that's fine */ }


 


      data = defaultData();


      if(migratedGroups) data.departments[0].groups = migratedGroups;


      await withTimeout(save(), 5000);


    }


  }catch(e){


    // Storage unavailable, timed out, or viewer isn't signed in — fall back to


    // a local (non-synced) copy so the checklist still loads instead of spinning forever.


    usingStorage = false;


    data = defaultData();


  }


  document.getElementById('titleDisplay').textContent = data.title;


  setupTitleEditing();


  render();


  if(!usingStorage){


    const note = document.querySelector('.note');


    note.innerHTML = '⚠️ Working offline — changes here won\'t sync to your team. Open the published link while signed in to Claude to sync.';


  }


}


 


load();
